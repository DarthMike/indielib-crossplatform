382  bool isCircleToCircleCollision(IND_Vector2 &pP1,
383  int pRadius1,
384  IND_Vector2 &pP2,
385  int pRadius2) {
386  // h^2 = x^2 + y^2 Pythagoras :D
387 
388  // x^2
389  double mDeltaXSquared = pP1._x - pP2._x;
390  mDeltaXSquared *= mDeltaXSquared;
391 
392  // y^2
393  double mDeltaYSquared = pP1._y - pP2._y;
394  mDeltaYSquared *= mDeltaYSquared;
395 
396  // Adding radius^2
397  double mSumRadiiSquared = pRadius1 + pRadius2;
398  mSumRadiiSquared *= mSumRadiiSquared;
399 
400  // If that result is greater than h^2, so less than x^2 + y^2
401  // then there is a collision
402  if (mDeltaXSquared + mDeltaYSquared <= mSumRadiiSquared)
403  return 1;
404  else
405  return 0;
406  }

419  bool isCircleToTriangleCollision(IND_Vector2 &pPCenter,
420  int pRadius,
421  IND_Vector2 &pA2,
422  IND_Vector2 &pB2,
423  IND_Vector2 &pC2) {
424  // Check if circle center inside the triangle
425  if (isPointInsideTriangle(pPCenter, pA2, pB2, pC2)) return 1;
426 
427  // Check the distance of the circle center to the 3 triangle segments
428  if (pointToLineDistance(pA2, pB2, pPCenter, 1) < pRadius) return 1;
429  if (pointToLineDistance(pB2, pC2, pPCenter, 1) < pRadius) return 1;
430  if (pointToLineDistance(pC2, pA2, pPCenter, 1) < pRadius) return 1;
431 
432  return 0;
433  }

444  static bool isPointInsideTriangle(IND_Vector2 &p,
445  IND_Vector2 &a,
446  IND_Vector2 &b,
447  IND_Vector2 &c) {
448  // Compute vectors
449  IND_Vector2 v0 = c - a;
450  IND_Vector2 v1 = b - a;
451  IND_Vector2 v2 = p - a;
452 
453  // Compute dot products
454  float dot00 = v0.dotProduct(v0);
455  float dot01 = v0.dotProduct(v1);
456  float dot02 = v0.dotProduct(v2);
457  float dot11 = v1.dotProduct(v1);
458  float dot12 = v1.dotProduct(v2);
459 
460  // Compute barycentric coordinates
461  float invDenom = 1 / (dot00 * dot11 - dot01 * dot01);
462  float u = (dot11 * dot02 - dot01 * dot12) * invDenom;
463  float v = (dot00 * dot12 - dot01 * dot02) * invDenom;
464 
465  // Check if point is in triangle
466  return (u > 0) && (v > 0) && (u + v < 1);
467  }

478  double pointToLineDistance(IND_Vector2 &pA,
479  IND_Vector2 &pB,
480  IND_Vector2 &pC,
481  bool pIsSegment) {
482  IND_Vector2 ab (pB._x - pA._x, pB._y - pA._y);
483  IND_Vector2 ac (pC._x - pA._x, pC._y - pA._y);
484 
485  double mDist (ab.crossProduct(ac) / pA.distance(pB));
486 
487  if (pIsSegment) {
488  IND_Vector2 bc (pC._x - pB._x, pC._y - pB._y);
489  IND_Vector2 ba (pA._x - pB._x, pA._y - pB._y);
490 
491  float mDot1 (ab.dotProduct(bc));
492  if (mDot1 > 0.0f) {
493  return pB.distance(pC);
494  }
495  float mDot2 (ba.dotProduct(ac));
496  if (mDot2 > 0.0f) {
497  return pA.distance(pC);
498  }
499  }
500 
501  return fabs(mDist);
502  }

514  bool isSegmentIntersection(IND_Vector2 &a,
515  IND_Vector2 &b,
516  IND_Vector2 &c,
517  IND_Vector2 &d) {
518  float mGradAB, mGradCD, mYcptAB, mYcptCD, mInterceptX, mInterceptY;
519 
520  // In order to avoid divisions by zero
521  if (a._y == b._y)
522  b._y += 0.0001f;
523 
524  if (a._x == b._x)
525  b._x += 0.0001f;
526 
527  if (c._y == d._y)
528  d._y += 0.0001f;
529 
530  if (c._x == d._x)
531  d._x += 0.0001f;
532 
533  // Calculates the intersection between the two lines
534  mGradAB = (a._y - b._y) / (a._x - b._x);
535  mGradCD = (c._y - d._y) / (c._x - d._x);
536 
537  mYcptAB = a._y - a._x * mGradAB;
538  mYcptCD = c._y - c._x * mGradCD;
539  mInterceptX = (mYcptAB - mYcptCD) / (mGradCD - mGradAB);
540  mInterceptY = (mYcptAB - (mGradAB * mYcptCD) / mGradCD) / (1 - mGradAB / mGradCD);
541 
542  // Checking in the intersection is inside the segment
543  if (!((mInterceptX >= a._x && mInterceptX <= b._x) || (mInterceptX >= b._x && mInterceptX <= a._x)))
544  return 0;
545 
546  if (!((mInterceptY >= a._y && mInterceptY <= b._y) || (mInterceptY >= b._y && mInterceptY <= a._y)))
547  return 0;
548 
549  if (!((mInterceptX >= c._x && mInterceptX <= d._x) || (mInterceptX >= d._x && mInterceptX <= c._x)))
550  return 0;
551 
552  if (!((mInterceptY >= c._y && mInterceptY <= d._y) || (mInterceptY >= d._y && mInterceptY <= c._y)))
553  return 0;
554 
555  return 1;
556  }
